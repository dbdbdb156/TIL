SELECT A.ANIMAL_ID, A.NAME
FROM ANIMAL_INS AS A JOIN ANIMAL_OUTS AS B
ON A.ANIMAL_ID = B.ANIMAL_ID
WHERE B.DATETIME < A.DATETIME
ORDER BY A.DATETIME ASC

### 2. 오랜 기간 보호한 동물(1)
- 쿼리 :
SELECT A.NAME, A.DATETIME
FROM ANIMAL_INS AS A LEFT JOIN ANIMAL_OUTS AS B
ON A.ANIMAL_ID = B.ANIMAL_ID
WHERE B.ANIMAL_ID IS NULL
ORDER BY A.DATETIME ASC LIMIT 3

### 3. 보호소에서 중성화한 동물
- 쿼리 :
SELECT A.ANIMAL_ID, A.ANIMAL_TYPE , A.NAME
FROM ANIMAL_OUTS AS A LEFT JOIN ANIMAL_INS AS B
ON A.ANIMAL_ID = B.ANIMAL_ID
WHERE B.SEX_UPON_INTAKE='Intact Male' AND A.SEX_UPON_OUTCOME='Neutered Male'
OR B.SEX_UPON_INTAKE='Intact Female' AND A.SEX_UPON_OUTCOME='Spayed Female'
ORDER BY A.ANIMAL_ID ASC
